    TYPE_OTHERS_REPORT          => دیگران - گزارش
    TYPE_OTHERS_PHOTO           => دیگران - عکس
    TYPE_MEMORY                 => خاطره
    TYPE_TELEX_TEXT             => تلکس متنی
    TYPE_TELEX_PHOTO            => تلکس عکس
